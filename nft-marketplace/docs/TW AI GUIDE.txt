type AuctionEvent = {

  buyoutPrice?: string;
  startingPrice?: string;
  endTime?: number;
  bidCount?: number;
  minNextBid?: string;
};

export default function NftGrid() {
  const [page, setPage] = useState(1);
  const [limit, setLimit] = useState(24);
  const [sort, setSort] = useState("buyoutPrice:asc");
  const [filters, setFilters] = useState<{search?: string; minPrice?: string; maxPrice?: string}>({});

  const [events, setEvents] = useState<AuctionEvent[]>([]);
  const [metas, setMetas]   = useState<Record<string, any>>({});
  const MARKET = process.env.NEXT_PUBLIC_MARKETPLACE_ADDRESS!;
  const COLLECTION = process.env.NEXT_PUBLIC_NFT_COLLECTION_ADDRESS!;

  async function load() {
    const qs = new URLSearchParams({ page: String(page), limit: String(limit), sort });
    if (filters.minPrice) qs.set("minPrice", filters.minPrice);
    if (filters.maxPrice) qs.set("maxPrice", filters.maxPrice);
    const res = await fetch(`/api/auctions?${qs.toString()}`);
    const data = await res.json();

    const evts: AuctionEvent[] = data.events ?? data.items ?? [];
    // Optional: apply name search after metadata (below) if needed
    setEvents(evts);

    // fetch metadata for each (in parallel)
    const reqs = evts.map(e =>
      fetch(`/api/metadata?contract=${e.nftContract ?? COLLECTION}&tokenId=${e.tokenId}`)
        .then(r => r.json())
        .then(j => ({ key: `${e.nftContract}:${e.tokenId}`, meta: j }))
    );
    const results = await Promise.all(reqs);
    const map: Record<string, any> = {};
    results.forEach(r => { map[r.key] = r.meta; });
    setMetas(map);
  }

  useEffect(() => { load(); /* eslint-disable-next-line */ }, [page, limit, sort, JSON.stringify(filters)]);

  // (Optional) client-side name search using fetched metadata
  const rows = useMemo(() => {
    if (!filters.search) return events;

      const key = `${e.nftContract ?? COLLECTION}:${e.tokenId}`;
      const name = metas[key]?.name?.toLowerCase?.() ?? "";
      return name.includes(q);
    });
  }, [events, metas, filters.search]);

  return (

function NftCard({ auction, metadata }: { auction: AuctionEvent; metadata: any }) {

export function BidBuyoutSection({
  contractAddress,

  minBid,
  buyoutPrice,

  contractAddress: string;
  auctionId: bigint;
  minBid?: string;

Thirdweb Marketplace English auctions have built-in enforcement for outbid increments and buyout rules:

---

### Built-in Functionality:
- When creating an English auction, you specify:
  - `startBidAmount` (minimum for the first bid)
  - `buyoutPrice` (anyone can end the auction immediately by paying this)
  - `bidBufferBps` (**basis points**, e.g. 500 = 5%) required for each outbid

- **Outbid logic is enforced by contract**:  
  - First bid must meet or exceed `startBidAmount`
  - Each new bid must be at least the highest bid * (1 + bidBufferBps/10,000)
  - The contract will reject bids below this; you do NOT need to enforce this yourself

- **Example**: To require a 5% minimum outbid, set `bidBufferBps: 500` when creating an auction

---

### UI/SDK Implementation
- For your "Place Bid" button, just use `bidInAuction()` from the `thirdweb/extensions/marketplace` SDK; the contract will enforce the increment.
- For "Buyout", use `buyoutAuction()`.
- Display in your UI:  
  - `Next Minimum Bid` = highestBid ? highestBid * 1.05 : startBidAmount

"use client";

import { useState } from "react";
import { useAuction } from "thirdweb/react";
import {
  bidInAuction,
  buyoutAuction,
  getContract,
} from "thirdweb/extensions/marketplace";
import { base } from "thirdweb/chains";
import { Input } from "@/components/ui/input"; // shadcn input
import { Button } from "@/components/ui/button"; // shadcn button

const MARKETPLACE_ADDRESS = process.env.NEXT_PUBLIC_MARKETPLACE_ADDRESS!;
const CLIENT_ID = process.env.NEXT_PUBLIC_CLIENT_ID!;

interface BidBuyoutSectionProps {
  listingId: string | number; // mapped from tokenId
  onBid?: () => void;
  onBuyout?: () => void;
}

export function BidBuyoutSection({
  listingId,
  onBid,
  onBuyout,
}: BidBuyoutSectionProps) {
  const [bidAmount, setBidAmount] = useState("");
  const [loading, setLoading] = useState(false);
  // Init contract client
  const contract = getContract({
    clientId: CLIENT_ID,
    address: MARKETPLACE_ADDRESS,
    chain: base,
  });

  const { data: auction, isLoading: auctionLoading } = useAuction({
    contract,
    listingId,
  });

  // Calculate next minimum bid (built-in enforcement, but show to user upfront)
  let nextMinBid = "";
  if (auction && auction.currentBid) {
    nextMinBid = (
      BigInt(auction.currentBid) + // current highest bid in wei
      (BigInt(auction.currentBid) * BigInt(auction.bidBufferBps)) / 10_000n
    ).toString();
  } else if (auction) {
    nextMinBid = auction.minBid; // starting price, in wei
  }

  // Convert wei to ether for display
  function formatEth(val: string | number) {
    return Number(val) / 1e18;
  }

  // Place Bid Handler
  async function handleBid() {
    setLoading(true);
    try {
      await bidInAuction({ contract, listingId, bidAmount: bidAmount }); // bidAmount must be a string in wei
      if (onBid) onBid();
    } catch (e) {
      console.error("Bid failed", e);
      // Show error toast/alert as needed
    }
    setLoading(false);
  }

  // Buyout Handler
  async function handleBuyout() {
    setLoading(true);
    try {
      await buyoutAuction({ contract, listingId });
      if (onBuyout) onBuyout();
    } catch (e) {
      console.error("Buyout failed", e);
    }
    setLoading(false);
  }

  // Convert displayed bid input to wei
  function toWei(amount: string) {
    return (Number(amount) * 1e18).toFixed(0);
  }

  if (auctionLoading) return <div>Loading auction info...</div>;
  if (!auction) return <div>Listing not found or not an English auction.</div>;

  return (
    <div className="flex flex-col gap-2">
      <div>
        <Input
          type="number"
          min={formatEth(nextMinBid)}
          step="0.0001"
          value={bidAmount}
          onChange={(e) => setBidAmount(e.target.value)}
          placeholder={"Bid ETH..."}
        />
        <div className="text-xs mt-1 text-muted-foreground">
          Next minimum bid: <b>{formatEth(nextMinBid)} ETH</b>
        </div>
      </div>
      <div className="flex gap-2">
        <Button
          onClick={() => handleBid()}
          disabled={
            loading || !bidAmount || Number(bidAmount) < formatEth(nextMinBid)
          }
        >
          Place Bid
        </Button>
        <Button
          variant="secondary"
          onClick={() => handleBuyout()}
          disabled={loading || !auction.buyoutPrice}
        >
          Buyout{" "}
          {auction.buyoutPrice && `(${formatEth(auction.buyoutPrice)} ETH)`}
        </Button>
      </div>
    </div>
  );
}

This is a very common scaling challenge for NFT marketplaces with large, live auction inventories. Here's how to approach it:
________________________________________
1. Why Price Sorting is Hard
‚Ä¢	Marketplace contract holds live auction prices (min bid, current/highest bid, buyout).
‚Ä¢	There is no onchain function to return all price points in a single call; you need to fetch or compute them per-listing.
‚Ä¢	Insight API cannot return live auction prices, only NFT data/metadata.
‚Ä¢	Doing 7777 individual getAuction calls would exhaust RPC rate limits and is not practical in real-time.
________________________________________
2. How Major Marketplaces Solve This
‚Ä¢	Server-Side Indexing: Periodically fetch (in batches/parallelized) all auction prices offchain/in a job, store them in your own indexed/local DB.
‚Ä¢	Your API returns /auctions?page=X&sort=price_asc, using cached/synced prices for all 7777 auctions.
‚Ä¢	Rehydrate/crawl auction contract state every ~15 seconds (or on relevant auction events).
‚Ä¢	Frontend only hits your paginated API, never calling 7777 auctions directly from browser.
________________________________________
3. Concrete Solutions For You
‚Ä¢	Set up a backend ETL/cron or queue worker that:
‚Ä¢	Loops through all active auction listingIds
‚Ä¢	Calls getAuction for each, gets live price info
‚Ä¢	Index/sorts in your DB (or Redis, etc), exposing /auctions?sort=price_asc for the grid
‚Ä¢	Optionally, subscribe to auction events (bid placed, buyout, new listing) to trigger a refresh of changed listings only
‚Ä¢	React Query/Frontend: Fetch only from your API sorted by price ‚Äî never scan/sort all 7777 auctions at client side or from browser!
________________________________________
4. Best Practices
‚Ä¢	Paginate your sorted endpoint (e.g., 60 items/page, infinite scroll, etc).
‚Ä¢	Always keep a 'last updated' timestamp on each price so users know when data was refreshed.
‚Ä¢	Don‚Äôt try to sort in frontend after fetching pages ‚Äî send pre-sorted slices from backend.
‚Ä¢	For search/filter, execute that logic in your API or DB, not in client code.
________________________________________
5. Summary Table
Need	Solution	Tool/API used
Sort all by price	Index contract state on backend; cache/sync all 7777	Contract reads + DB
Live prices	getAuction per listing (backend only)	Thirdweb v5 SDK server-side
Minimize RPC	Batched, throttled fetch + event hooks	Backend job, not client
Sort/Filter UX	Paginate pre-sorted API	Custom /auctions endpoint

Here are battle-tested tips for handling rate limits, caching, and pagination in production web3 marketplaces:
________________________________________
1. Pagination
‚Ä¢	Always paginate API/results: never try to load >60‚Äì100 items at once.
‚Ä¢	Use cursor or page-number pagination from your data/API provider (Insight supports both).
‚Ä¢	Design your backend endpoints to mirror third-party limits‚Äîe.g., if Insight returns 40 per page, keep your own endpoints/pages matched.
2. Caching
‚Ä¢	Server-side cache: Store fetched data in Redis, memory, or a database with a short expiry (10‚Äì60s for hot auction/NFT data).
‚Ä¢	Browser cache/React Query: Use libraries like React Query to cache and deduplicate fetches per page/filters. Use staleTime to control refresh intervals‚Äîset lower values for auction data, higher for static metadata.
‚Ä¢	Invalidate/refresh: After user transaction (bid/buyout), invalidate only the relevant cache keys/pages.
3. Managing Rate Limits
‚Ä¢	Batch/Throttle: Never call more than your lowest API/RPC plan limit. If your vendor is 250 RPS, stay well under that (e.g., cap at 150‚Äì200 RPS).
‚Ä¢	Backoff on 429 errors: If you receive 429 Rate Limit errors, back off (wait) and retry slowly. Don‚Äôt hammer!
‚Ä¢	Keep all "contract reads" off the frontend‚Äîlet the backend/indexer handle sync and cache.
‚Ä¢	Prioritize updating "hot" (soon-to-close, most-viewed, active) auctions more often; update cold/expired auctions infrequently.
4. Tips & Tricks
‚Ä¢	Lazy Load: Only fetch next pages on scroll/pagination, not in advance.
‚Ä¢	Event-driven refresh: Subscribe to events (e.g., bid placed) to refresh only impacted listings.
‚Ä¢	Sort/filter on server, not client: Always apply price/rank sorts and search at the DB/API level.
‚Ä¢	Metrics Alerting: Monitor your RPS/chart over time; scale up infrastructure ahead of big drops.
________________________________________
In production: serve static or lightly dynamic pages from cache, and only re-fetch/sync what absolutely needs to be live‚Äîlet users trigger contract writes, but never show data from the chain ‚Äúlive‚Äù unless essential.
